programa programa3;

var k: entero; 

nula factorial(k: entero, producto: entero){
    {   
        si (k == 0){
            escribe(producto); 
        } sino {
            factorial(k-1, producto * k);
        };  
    }
};

nula suma(k: entero){
    var sum: entero;

    {
        sum = 0;
        
        mientras (k > 0) haz {
            sum = sum + k;
            k = k - 1;
        }; 

        escribe(sum);
    }
};

nula resta(a: entero, b: entero){
    {
        escribe(a - b);
    }
};

inicio {
    k = 1; 
    
    factorial(k + 3, 1);
    suma(5);
    resta(0, 10);
}

fin
