programa programa3;

var k: entero; 

nula factorial(k: entero){
    var producto: entero;

    {
        producto = 1;
        
        mientras (k > 0) haz {
            producto = producto * k;
            k = k - 1;
        }; 

        escribe(producto);
    }
};

nula suma(k: entero){
    var sum: entero;

    {
        sum = 0;
        
        mientras (k > 0) haz {
            sum = sum + k;
            k = k - 1;
        }; 

        escribe(sum);
    }
};

nula resta(a: entero, b: entero){
    {
        escribe(a - b);
    }
};

inicio {
    k = 1;
    
    factorial(k + 2);
    suma(5);
    resta(0, 10);
}
